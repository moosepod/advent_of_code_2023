def HASH(s: str) -> int:
    v = 0
    for c in s:
        v += ord(c)
        v *= 17
        v = v % 256
        
    return v

import re
def star_1(path: str) -> int:
  v = 0
  with open(path,'r') as f:
      v += sum([HASH(s) for s in f.read().strip().split(',')])

  return v
  
print("Star 1:",star_1("input.txt"))
